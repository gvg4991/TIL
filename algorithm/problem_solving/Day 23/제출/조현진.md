   def B(x):
       global b
       c=0
       q=2**(L-1)
       while q>0:
           b+=int(x[c])*q
           c+=1
           q//=2
   def sam(x):
       global S
       while len(S)!=LL:
           S+=str(x%3)
           x//=3

   for n in range(T):
       N=input()
       M=input()

       LL=len(M)
       for x in range(L):
           b=0
           S=''
           B(N)
           if N[x]=='0':
               b+=2**(L-x-1)

               b-=2**(L-x-1)
           sam(b)
           S=S[::-1]
           cnt=0
