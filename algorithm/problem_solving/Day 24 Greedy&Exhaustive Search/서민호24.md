def selection(start,end):
    if start == end:
        return
    else:
        target = min(datas[start:end])
        idx = datas.index(target)
        datas[idx] = datas[start]
        datas[start] = target
    selection(start+1,end)

datas = list(map(int,' '.join(map(str,input())).split()))
selection(0,len(datas))
print(datas)
